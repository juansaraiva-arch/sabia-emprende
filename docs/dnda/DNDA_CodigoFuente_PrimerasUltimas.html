339export function computeOxigeno(r: FinancialRecord) {
340  const diasCalle =
341    r.revenue > 0 ? (r.accounts_receivable / r.revenue) * 30 : 0;
342  const diasInventario =
343    r.cogs > 0 ? (r.inventory / r.cogs) * 30 : 0;
344  const diasProveedor =
345    r.cogs > 0 ? (r.accounts_payable / r.cogs) * 30 : 0;
346  const ccc = diasCalle + diasInventario - diasProveedor;
347
348  const totalLiquido = r.cash_balance + r.accounts_receivable;
349  const pasivoCorto = r.accounts_payable + r.bank_debt;
350  const pruebaAcida = pasivoCorto > 0 ? totalLiquido / pasivoCorto : 0;
351
352  const dineroAtrapado = r.accounts_receivable + r.inventory;
353
354  const cascada = computeCascada(r);
355  const coberturaBancaria =
356    r.interest_expense > 0 ? cascada.ebitda / r.interest_expense : 10;
357
358  return {
359    diasCalle,
360    diasInventario,
361    diasProveedor,
362    ccc,
363    pruebaAcida,
364    dineroAtrapado,
365    coberturaBancaria,
366    totalLiquido,
367    pasivoCorto,

292/**
293 * Cierra un reporte (accion irreversible).
294 * Una vez cerrado, el reporte queda como snapshot inmutable
295 * y no se puede modificar ni eliminar.
296 */
297export function cerrarReporte(reporteId: string): ReporteMensual | null {
298  const all = readReportes();
299  const idx = all.findIndex((r) => r.id === reporteId);
300  if (idx < 0) return null;
301
302  const reporte = all[idx];
303  if (reporte.estado === "cerrado") return reporte; // ya cerrado
304
305  all[idx] = {
306    ...reporte,
307    estado: "cerrado",
308    cerrado_en: new Date().toISOString(),
309  };
310
311  writeReportes(all);
312  return all[idx];
313}

315/**
316 * Elimina un reporte en estado borrador.
317 * Los reportes cerrados no se pueden eliminar.
318 */
319export function eliminarBorrador(reporteId: string): boolean {
320  const all = readReportes();
321  const idx = all.findIndex((r) => r.id === reporteId);
322  if (idx < 0) return false;
323
324  const reporte = all[idx];
325  if (reporte.estado === "cerrado") return false;
326
327  all.splice(idx, 1);
328  writeReportes(all);
329  return true;
330}
